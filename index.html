const slides = track.querySelectorAll('.slide'); let slideIndex = 0; const slideWidth = () => slides[0].offsetWidth + 16; // gap dahil // Ok butonları ile kaydırma prevBtn.addEventListener('click', () => { slideIndex = (slideIndex - 1 + slides.length) % slides.length; track.scrollTo({ left: slideWidth() * slideIndex, behavior: 'smooth' }); }); nextBtn.addEventListener('click', () => { slideIndex = (slideIndex + 1) % slides.length; track.scrollTo({ left: slideWidth() * slideIndex, behavior: 'smooth' }); }); // Otomatik kayma let autoSlide = setInterval(() => { slideIndex = (slideIndex + 1) % slides.length; track.scrollTo({ left: slideWidth() * slideIndex, behavior: 'smooth' }); }, 3000); // 3 saniyede bir kayar // Otomatik kaymayı durdurma (parmakla dokununca) track.addEventListener('mouseenter', () => clearInterval(autoSlide)); track.addEventListener('mouseleave', () => { autoSlide = setInterval(() => { slideIndex = (slideIndex + 1) % slides.length; track.scrollTo({ left: slideWidth() * slideIndex, behavior: 'smooth' }); }, 3000);
